numero = float(input("Digite um número para calcular a raiz quadrada: "))

else:

    7 Escreva um programa que calcule a área de uma circunferência. O usuário deve digitar o valor do raio e ao final o programa deverá mostrar na tela a área da circunferência. Use a fórmula: área=pi*r² , em que pi é uma constante e r o raio da circunferência. 

Dica: você pode usar a biblioteca math para pegar a constante pi. 
from math import pi 

import math
raio = float(input("Digite o valor do raio da circunferência: "))
area = math.pi * (raio ** 2)
print(f"A área da circunferência é: {area:.2f}")

    8 Escreva um programa que receba o nome, sobrenome e idade do usuário e apresente a seguinte mensagem na tela: 

"Seja bem-vindo [nome] [sobrenome]." 

"Você possui [idade] anos de idade." 

No campo nome e sobrenome utilize os métodos strip() e title(). Lembre-se que o primeiro método permite remover os espaços antes e depois da string, enquanto que o último permite colocar a string no formato titlecased (capitaliza a string). 

nome = input("Digite seu nome: ").strip().title()
sobrenome = input("Digite seu sobrenome: ").strip().title()
idade = int(input("Digite sua idade: "))
print(f"\nSeja bem-vindo(a), {nome} {sobrenome}.")
print(f"Você possui {idade} anos de idade.")

    9 Escreva um programa que peça o nome e a idade do usuário. Caso a idade do usuário seja maior ou igual a 18 anos apresente a seguinte mensagem: "Seja bem-vindo ao nosso site [nome]!"; caso contrário, apresente a seguinte mensagem: "Você não pode acessar nosso site [nome].". 

def verificar_acesso():
  nome = input("Digite seu nome: ")
  idade = int(input("Digite sua idade: "))
  if idade >= 18:
    print(f"Seja bem-vindo ao nosso site {nome}!")
  else:
    print(f"Você não pode acessar nosso site {nome}.")
verificar_acesso()
